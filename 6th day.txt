array reverse if array size is greater then 3:

#include <iostream>

using namespace std;

int main()
{
    int temp,len, a[6]={1,2,3,4,5,6};
    len=6/2;
    for(int i=0;i<len;i++){
        temp=a[i];
        a[i]=a[5-i];
        a[5-i]=temp;
    }
   for(int i=0; i<6;i++){
       cout<<a[i];
   }
    return 0;
}

for all size array reverse we use :

#include <iostream>

using namespace std;

int main()
{
    int start=0,end=2, a[3]={1,2,3};
    while(start<end){
        swap(a[start],a[end]);
        ++start;
        --end;
    }
   for(int i=0; i<3;i++){
       cout<<a[i];
   }
    return 0;
}

fibonnic series:

#include <iostream>

using namespace std;

int main()
 { 
     int temp=0,first=0,second=1;
    cout<<"fibonnic series:"<<first<<" "<<second<<" ";
  
    
    for(int i=0; i<6;i++){
  
        temp=first+second;
        first=second;
        second =temp;
        cout<<temp<<" ";
    }
    return 0;
}
