decimal to binary:

#include <iostream>

using namespace std;

int main()
{ 
    int n,rem,ans,mul=1;
    cout<<"please enter input to convert into a binary number:";
    cin>>n;
    while(n>0){
        rem=n%2;
        n=n/2;
        ans= rem*mul+ans;
        mul=mul*10;
        
    }
    cout<<"binary number converted:"<<ans;
}

binary to decimal:

#include <iostream>

using namespace std;

int main()
{ 
    int n,rem,ans,mul=1;
    cout<<"please enter input to convert into a decimal number:";
    cin>>n;
    while(n>0){
        rem=n%10;
        n=n/10;
        ans= mul*rem+ans;
        mul=2*mul;
        
    }
    cout<<"binary number converted to decimal:"<<ans;
}
