Bishop moves in chess 8*8:

#include <iostream>

using namespace std;

int main()
{
  int A,B,count=0;
  cout << "ENTER VALUE OF COORDINATES OF A AND B:" << endl;
 cin>>A>>B;
 count+=min(8-A,8-B);
 count+=min(8-A,B-1);
 count+=min(A-1,B-1);
 count+=min(A-1,8-B);
 cout<<count;
    return 0;
}

linear search :



#include <iostream>

using namespace std;

int main()
{
  int B ,c=0,arr[]={2,3,4,5,6,7,8,9,10};
  
  cout << "ENTER VALUE search:" << endl;
 cin>>B;
 for(int i=0; i<sizeof(arr);i++){
     if(arr[i]==B){
         cout<<"MATCHED";
         c=0;
         break;
     }else{
         c=1;
     }
 }
 if(c==1){
     cout<<"not matched";
 }
    return 0;
}
