
#include <iostream>

using namespace std;

int main()
{
    int num,rep,rem,ans=0;
    cout<<"enter a number to check whether it is a palindrone or not:";
    cin>>num;
    rep=num;
    while(num>0){
        rem= num%10;
        num/=10;
        ans=ans*10+rem;
    }
    cout<<rep<<" "<<ans;
    if(rep==ans){
        cout<<"number is palindrone.";
    }else{
        cout<<"not a palindrone.";
    }
    return ans;
}
